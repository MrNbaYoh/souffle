132 class LVMIndirectRelation : public LVMRelation {
133 public:
134  LVMIndirectRelation(size_t relArity, const MinIndexSelection* orderSet, std::string& relName,
135  std::vector<std::string>& attributeTypes)
136  : LVMRelation(relArity, orderSet, relName, attributeTypes) {
137  for (auto& order : orderSet->getAllOrders()) {
138  indices.push_back(LVMIndex(order));
139  }
140  }
141 
142  /** Insert tuple */
143  void insert(const RamDomain* tuple) override {
144  assert(tuple);
145 
146  // make existence check
147  if (exists(tuple)) {
148  return;
149  }
150 
151  int blockIndex = num_tuples / (BLOCK_SIZE / arity);
152  int tupleIndex = (num_tuples % (BLOCK_SIZE / arity)) * arity;
153 
154  if (tupleIndex == 0) {
155  blockList.push_back(std::make_unique<RamDomain[]>(BLOCK_SIZE));
156  }
157 
158  RamDomain* newTuple = &blockList[blockIndex][tupleIndex];
159  for (size_t i = 0; i < arity; ++i) {
160  newTuple[i] = tuple[i];
161  }
162 
163  // update all indexes with new tuple
164  for (auto& cur : indices) {
165  cur.insert(newTuple);
166  }
167 
168  // increment relation size
169  num_tuples++;
170  }
171 
172  /** Merge another relation into this relation */
173  void insert(const LVMRelation& other) override {
174  assert(getArity() == other.getArity());
175  for (const auto& cur : other) {
176  insert(cur);
177  }
178  }
179 
180  /** Purge table */
181  void purge() override {
182  blockList.clear();
183  for (auto& cur : indices) {
184  cur.purge();
185  }
186  num_tuples = 0;
187  }
188 
189  /** check whether a tuple exists in the relation */
190  bool exists(const RamDomain* tuple) const override {
191  LVMIndex* index = getIndex(getTotalIndexKey());
192  return index->exists(tuple);
193  }
194 
195  /** Iterator for relation, uses full-order index as default */
196  iterator begin() const override {
197  return indices[0].begin();
198  }
199 
200  iterator end() const override {
201  return indices[0].end();
202  }
203 
204  /** Return range iterator */
205  std::pair<iterator, iterator> lowerUpperBound(
206  const RamDomain* low, const RamDomain* high, size_t indexPosition) const override {
207  auto idx = this->getIndexByPos(indexPosition);
208  return idx->lowerUpperBound(low, high);
209  }
210 
211  /** Extend tuple */
212  std::vector<RamDomain*> extend(const RamDomain* tuple) override {
213  std::vector<RamDomain*> newTuples;
214 
215  // A standard relation does not generate extra new knowledge on insertion.
216  newTuples.push_back(new RamDomain[2]{tuple[0], tuple[1]});
217 
218  return newTuples;
219  }
220 
221  /** Extend relation */
222  void extend(const LVMRelation& rel) override {}
223 
224  /** get index for a given search signature. Order are encoded as bits for each column */
225  LVMIndex* getIndex(const SearchSignature& col) const {
226  // Special case in provenance program, a 0 searchSignature is considered as a full search
227  if (col == 0) {
228  return getIndex(getTotalIndexKey());
229  }
230  return getIndexByPos(orderSet->getLexOrderNum(col));
231  }
232 
233  /** get index for a given order. Order are encoded as bits for each column */
234  LVMIndex* getIndexByPos(int idx) const {
235  return &indices[idx];
236  }
237 
238  /** Obtains a full index-key for this relation */
239  SearchSignature getTotalIndexKey() const {
240  return (1 << (getArity())) - 1;
241  }
242 
243 private:
244  /** Size of blocks containing tuples */
245  static const int BLOCK_SIZE = 1024;
246 
247  std::deque<std::unique_ptr<RamDomain[]>> blockList;
248 
249  /** List of indices */
250  mutable std::vector<LVMIndex> indices;
251 };

342  /** Insert tuple */
343  void insert(const RamDomain* tuple) override {
344  // TODO: (pnappa) an eqrel check here is all that appears to be needed for implicit additions
345  // TODO: future optimisation would require this as a member datatype
346  // brave soul required to pass this quest
347  // // specialisation for eqrel defs
348  // std::unique_ptr<binaryrelation> eqreltuples;
349  // in addition, it requires insert functions to insert into that, and functions
350  // which allow reading of stored values must be changed to accommodate.
351  // e.g. insert => eqRelTuples->insert(tuple[0], tuple[1]);
352 
353  // for now, we just have a naive & extremely slow version, otherwise known as a O(n^2) insertion
354  // ):
355 
356  for (auto* newTuple : extend(tuple)) {
357  LVMIndirectRelation::insert(newTuple);
358  delete[] newTuple;
359  }
360  }
361 
362  /** Find the new knowledge generated by inserting a tuple */
363  std::vector<RamDomain*> extend(const RamDomain* tuple) override {
364  std::vector<RamDomain*> newTuples;
365 
366  newTuples.push_back(new RamDomain[2]{tuple[0], tuple[0]});
367  newTuples.push_back(new RamDomain[2]{tuple[0], tuple[1]});
368  newTuples.push_back(new RamDomain[2]{tuple[1], tuple[0]});
369  newTuples.push_back(new RamDomain[2]{tuple[1], tuple[1]});
370 
371  std::vector<const RamDomain*> relevantStored;
372  for (const RamDomain* vals : *this) {
373  if (vals[0] == tuple[0] || vals[0] == tuple[1] || vals[1] == tuple[0] || vals[1] == tuple[1]) {
374  relevantStored.push_back(vals);
375  }
376  }
377 
378  for (const auto vals : relevantStored) {
379  newTuples.push_back(new RamDomain[2]{vals[0], tuple[0]});
380  newTuples.push_back(new RamDomain[2]{vals[0], tuple[1]});
381  newTuples.push_back(new RamDomain[2]{vals[1], tuple[0]});
382  newTuples.push_back(new RamDomain[2]{vals[1], tuple[1]});
383  newTuples.push_back(new RamDomain[2]{tuple[0], vals[0]});
384  newTuples.push_back(new RamDomain[2]{tuple[0], vals[1]});
385  newTuples.push_back(new RamDomain[2]{tuple[1], vals[0]});
386  newTuples.push_back(new RamDomain[2]{tuple[1], vals[1]});
387  }
388 
389  return newTuples;
390  }
391  /** Extend this relation with new knowledge generated by inserting all tuples from a relation */
392  void extend(const LVMRelation& rel) override {
393  std::vector<RamDomain*> newTuples;
394  // store all values that will be implicitly relevant to the those that we will insert
395  for (const auto* tuple : rel) {
396  for (auto* newTuple : extend(tuple)) {
397  newTuples.push_back(newTuple);
398  }
399  }
400  for (const auto* newTuple : newTuples) {
401  LVMIndirectRelation::insert(newTuple);
402  delete[] newTuple;
403  }
404  }
405 };
