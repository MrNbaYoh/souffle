32  struct comparator {

38  /* comparison function */
39  int operator()(const RamDomain* x, const RamDomain* y) const {
40  for (int i : order) {
41  if (x[i] < y[i]) {
42  return -1;
43  }
44  if (x[i] > y[i]) {
45  return 1;
46  }
47  }
48  return 0;
49  }

53  return operator()(x, y) < 0;
54  }

56  /* equal comparison */
57  bool equal(const RamDomain* x, const RamDomain* y) const {
58  for (int i : order) {
59  if (x[i] != y[i]) {
60  return false;

63  return true;
64  }
